## User is present
<Story name="User is present">
  <Grid container direction="row" alignItems="center" spacing={2}>
    {
      forEachSize((sizeName) => {
        const onClick = actions('onClick')
        const displayUserNameTooltip = boolean('Display user name tooltip', true)
        const disabled = boolean('Disabled', false)
        return (
          <UserAvatarWithPlaceholder
            size={sizeName}
            user={makeSampleUser(false, false)}
            disabled={disabled}
            displayUserNameOnHover={displayUserNameTooltip}
            variant="grey"
            {...onClick}
          />
        )
      })
    }
  </Grid>
