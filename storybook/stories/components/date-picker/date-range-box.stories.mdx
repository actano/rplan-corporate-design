import { Meta, Story } from '@storybook/addon-docs/blocks'

  title="Components/Date Pickers/Date Range Picker"
  decorators={decorators}

export const TemplateDateRangePicker = (args) => {
  const onAcceptStart = actions('onAcceptStart')
  const onAcceptEnd = actions('onAcceptEnd')
